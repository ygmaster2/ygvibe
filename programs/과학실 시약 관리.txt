// 2. 관리자 비밀번호 확인
function verifyAdminPin(pin) {
  return pin === "17752";

// 3. 사용자용 시약 목록 가져오기
function getReagentList(category) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(category);
  if (!sheet) return [];
  var lastRow = sheet.getLastRow();
  if (lastRow < 3) return [];
  
  var data = sheet.getRange(3, 2, lastRow - 2, 7).getValues(); 
  var list = [];
  for (var i = 0; i < data.length; i++) {
    if (data[i][0] !== "" && data[i][1] !== "") { // 순번과 이름이 있는 경우만
      list.push({
        id: data[i][0],      // B열
        name: data[i][1],    // C열
        remain: data[i][6],  // H열
        rowIndex: i + 3
      });
    }
  }
  return list;
}

// 4. 사용 기록 제출 및 잔량 차감
function submitUsage(date, category, reagentId, reagentName, amount, user) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var logSheet = ss.getSheetByName('사용기록');
  var categorySheet = ss.getSheetByName(category);
  
  logSheet.appendRow([date, reagentId, reagentName, amount, user]);
  
  var data = categorySheet.getRange(3, 2, Math.max(1, categorySheet.getLastRow() - 2), 7).getValues();
  for (var i = 0; i < data.length; i++) {
    if (data[i][0] == reagentId) {
      var currentRow = i + 3;
      var newRemain = Number(data[i][6]) - Number(amount);
      categorySheet.getRange(currentRow, 8).setValue(newRemain);
      break;
    }
  }
  return "기록되었습니다.";
}

// 5. 관리자 데이터 로드 (문제 1, 2 해결)
function getSheetDataForAdmin(sheetName) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(sheetName);
  var rawData = [];
  var lastRow = sheet.getLastRow();
  
  if (sheetName === '사용기록') {
    // 문제 1: 3행부터 5열까지 가져오기 (A3:E3가 헤더)
    rawData = sheet.getRange(3, 1, Math.max(1, lastRow - 2), 5).getDisplayValues();
  } else {
    // B2부터 H열까지 (2행이 헤더)
    rawData = sheet.getRange(2, 2, Math.max(1, lastRow - 1), 7).getDisplayValues();
  }

  // 문제 2: 내용이 없는 행(공란) 생략 필터링
  var filteredData = rawData.filter(function(row, index) {
    if (index === 0) return true; // 헤더는 유지
    return row.join("").trim() !== ""; // 행 전체가 비어있지 않은 경우만 포함
  });

  return filteredData;
}

// 6. 관리자 모드: 시약 추가 기능 (문제 3 해결)
function addReagentByAdmin(category, reagentData) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(category);
  if (!sheet) return "시트를 찾을 수 없습니다.";

  // 양식: A(분류), B(순번), C(시약명), D(화합물), E(용량), F(날짜), G(초기량), H(잔량)
  sheet.appendRow([
    category,
    reagentData.id,
    reagentData.name,
    reagentData.compound,
    reagentData.capacity,
    reagentData.date,
    reagentData.initial,
    reagentData.initial // 초기 잔량은 초기량과 동일하게 세팅
  ]);
  return category + " 카테고리에 추가되었습니다.";
}

// 7. 수정 및 삭제
function updateReagentByAdmin(sheetName, rowIndex, newAmount, isDelete) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(sheetName);
  if (isDelete) {
    sheet.deleteRow(rowIndex);
    return "삭제되었습니다.";
  } else {
    var col = (sheetName === '사용기록') ? 4 : 8;
    sheet.getRange(rowIndex, col).setValue(newAmount);
    return "수정되었습니다.";
  }
}

// 관리자 데이터 로드 (핵심 수정: 실제 행 번호 포함)
function getSheetDataForAdmin(sheetName) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(sheetName);
  var lastRow = sheet.getLastRow();
  var startRow = (sheetName === '사용기록') ? 3 : 2;
  var numCols = (sheetName === '사용기록') ? 5 : 7;
  var startCol = (sheetName === '사용기록') ? 1 : 2;
  
  if (lastRow < startRow) return [];
  var values = sheet.getRange(startRow, startCol, lastRow - startRow + 1, numCols).getDisplayValues();
  
  var filteredData = [];
  for (var i = 0; i < values.length; i++) {
    var row = values[i];
    var actualRowIndex = startRow + i; // 시트상의 실제 행 번호 계산
    
    // 헤더이거나 내용이 있는 행만 포함
    if (i === 0 || row.join("").trim() !== "") {
      var rowWithIndex = row.slice(); 
      rowWithIndex.push(actualRowIndex); // 배열의 마지막에 실제 인덱스 숨겨서 전달
      filteredData.push(rowWithIndex);
    }
  }
  return filteredData;
}

  <script>
    document.getElementById('use-date').valueAsDate = new Date();
    var currentReagentList = [];
    var spreadsheetUrl = "";

    // PIN 모달 제어 함수
    function showPinModal() {
      document.getElementById('pin-modal').style.display = 'block';
      document.getElementById('pin-input').focus();
    }
    function closePinModal() {
      document.getElementById('pin-modal').style.display = 'none';
      document.getElementById('pin-input').value = '';
    }

    function checkAdminPin() {
      var pin = document.getElementById('pin-input').value;
      google.script.run.withSuccessHandler(function(ok){
        if(ok){ 
          closePinModal();
          document.getElementById('user-view').style.display='none'; 
          document.getElementById('admin-view').style.display='block'; 
          loadAdminData('사용기록');
          // 관리자 진입 시 시트 URL 미리 가져오기
          google.script.run.withSuccessHandler(function(url){ spreadsheetUrl = url; }).getSpreadsheetUrl();
        } else {
          alert("PIN 번호가 일치하지 않습니다.");
          document.getElementById('pin-input').value = '';
        }
      }).verifyAdminPin(pin);
    }

    function openSpreadsheet() {
      if(spreadsheetUrl) window.open(spreadsheetUrl, '_blank');
      else alert("주소를 불러오는 중입니다. 잠시 후 다시 클릭해주세요.");
    }

    // 기존 함수들 (변경 없음)
    function loadReagents() {
      var cat = document.getElementById('category').value;
      var sel = document.getElementById('reagent-select');
      sel.innerHTML = '<option value="">데이터 불러오는 중...</option>';
      google.script.run.withSuccessHandler(function(list) {
        currentReagentList = list;
        sel.innerHTML = '<option value="">시약 선택 (순번, 이름)</option>';
        list.forEach(function(item) {
          sel.innerHTML += `<option value="${item.id}">${item.id}, ${item.name} (잔량: ${item.remain})</option>`;
        });
      }).getReagentList(cat);
    }

    function submitData() {
      var date = document.getElementById('use-date').value, cat = document.getElementById('category').value, id = document.getElementById('reagent-select').value, amt = document.getElementById('amount').value, user = document.getElementById('user-info').value;
      if(!id || !amt || !user) return alert("모든 항목을 입력해주세요.");
      var name = currentReagentList.find(i => i.id === id).name;
      google.script.run.withSuccessHandler(function(m){ alert(m); loadReagents(); }).submitUsage(date, cat, id, name, amt, user);
    }

    function addNewReagent() {
      var cat = document.getElementById('add-cat').value;
      var data = { id: document.getElementById('add-id').value, name: document.getElementById('add-name').value, compound: document.getElementById('add-comp').value, capacity: document.getElementById('add-cap').value, initial: document.getElementById('add-init').value, date: new Date().toLocaleDateString() };
      if(!data.id || !data.name) return alert("순번과 시약명은 필수입니다.");
      google.script.run.withSuccessHandler(function(m){ alert(m); loadAdminData(cat); }).addReagentByAdmin(cat, data);
    }

    function openTab(evt, name) {
      var links = document.getElementsByClassName("tablinks");
      for (var i=0; i<links.length; i++) links[i].className = links[i].className.replace(" active", "");
      var contents = document.getElementsByClassName("tabcontent");
      for (var i=0; i<contents.length; i++) contents[i].style.display = "none";
      document.getElementById(name).style.display = "block";
      evt.currentTarget.className += " active";
      loadAdminData(name);
    }

    function loadAdminData(name) {
      var div = document.getElementById(name);
      div.innerHTML = "<p style='text-align:center;'>연동 중...</p>";
      google.script.run.withSuccessHandler(function(data){
        var html = "<table>";
        data.forEach(function(row, i){
          var actualRow = row.pop();
          html += "<tr>";
          row.forEach(function(cell){ html += (i===0) ? `<th>${cell}</th>` : `<td>${cell}</td>`; });
          if(i===0) html += "<th>관리</th>";
          else {
            html += `<td>
              <button class='action-btn btn-edit' onclick='editRow("${name}",${actualRow})'>수정</button>
              <button class='action-btn btn-delete' onclick='deleteRow("${name}",${actualRow})'>삭제</button>
            </td>`;
          }
          html += "</tr>";
        });
        div.innerHTML = html + "</table>";
      }).getSheetDataForAdmin(name);
    }

    function editRow(name, row) {
      var val = prompt("수정할 수치를 입력하세요:");
      if(val) google.script.run.withSuccessHandler(function(m){ alert(m); loadAdminData(name); }).updateReagentByAdmin(name, row, val, false);
    }

    function deleteRow(name, row) {
      if(confirm("이 행을 삭제하시겠습니까? 시트에서도 삭제됩니다.")) {
        google.script.run.withSuccessHandler(function(m){ alert(m); loadAdminData(name); }).updateReagentByAdmin(name, row, 0, true);
      }
    }
  </script>
