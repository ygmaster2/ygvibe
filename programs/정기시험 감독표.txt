function getSheetData() {
  try {
    var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
    var title = sheet.getRange("A1").getDisplayValue();
    
    // F1 셀 시간 확인
    var openTimeCell = sheet.getRange("F1").getValue();
    var isOpened = true;
    var openTimeStr = "";

    if (openTimeCell !== "" && openTimeCell !== null) {
      var openDate = new Date(openTimeCell);
      if (!isNaN(openDate.getTime())) {
        var now = new Date();
        if (now < openDate) {
          isOpened = false;
          var y = openDate.getFullYear();
          var m = openDate.getMonth() + 1;
          var d = openDate.getDate();
          var h = openDate.getHours();
          var min = openDate.getMinutes();
          var minFormatted = min < 10 ? '0' + min : min;
          openTimeStr = y + "년 " + m + "월 " + d + "일 " + h + "시 " + minFormatted + "분";
          
          // 공개 전이면 남은 시간 계산을 위해 openTimestamp(밀리초)를 같이 보냄
          return { 
            isOpened: false, 
            title: title, 
            openTimeStr: openTimeStr, 
            openTimestamp: openDate.getTime() 
          };
        }
      }
    }

    // E1 셀의 값을 확인하여 3교시인지 4교시인지 판별 (A1:F 범위로 확장)
    var periodTypeCell = sheet.getRange("E1").getDisplayValue();
    var maxPeriod = (periodTypeCell.indexOf("4교") !== -1) ? 4 : 3;
    
    // 4교시까지 고려하여 범위를 A1:F로 확장하여 가져옴
    var data = sheet.getRange("A1:F").getDisplayValues();
    
    return { isOpened: true, title: title, data: data, maxPeriod: maxPeriod };
    
  } catch (error) {
    return { isError: true, message: error.message };
  }
}

// [추가된 기능] 시트의 마지막 수정 시간을 가져오는 함수
function getLastModifiedTime() {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var file = DriveApp.getFileById(ss.getId());
  return file.getLastUpdated().getTime();
}

    var timerInterval;

      document.getElementById('mainContent').style.display = 'none';
      
      google.script.run
        .withSuccessHandler(buildUI)
        .withFailureHandler(function(error) {
          showError("서버 연결 오류: " + error.message);
        })
        .getSheetData();
    }

      document.getElementById('loading').style.display = 'none';
      var errorBox = document.getElementById('errorBox');
      errorBox.style.display = 'block';
      errorBox.innerHTML = "문제가 발생했습니다!<br><br>" + msg;
    }

    function buildUI(response) {
      document.getElementById('loading').style.display = 'none';

      if (!response || response.isError) {

        return;
      }

      // 시간표 공개 전 (타이머 작동)
      if (response.isOpened === false) {
        document.getElementById('closedMessage').style.display = 'block';
        document.getElementById('closedTitle').innerText = response.title;

    // [추가된 기능] 서버(시트)의 마지막 수정 시간을 확인
    function pollForUpdates() {
      google.script.run
        .withSuccessHandler(function(serverTime) {
          if (currentModifiedTime === null) {
            currentModifiedTime = serverTime; // 처음 켤 때 기준 시간 세팅
          } else if (serverTime > currentModifiedTime) {
            // 시트가 수정됨! 새로고침 버튼 띄우기
            document.getElementById('updateAlertBtn').style.display = 'block';
          }
        })
        .getLastModifiedTime();
    }

    // [추가된 기능] 새로고침 버튼을 눌렀을 때 실행
    function reloadTableData() {
      document.getElementById('updateAlertBtn').style.display = 'none'; // 버튼 숨기기
      currentModifiedTime = null; // 시간 초기화
      fetchData(); // 데이터 다시 불러오기
    }

    function startCountdown(targetTime) {
      if (timerInterval) clearInterval(timerInterval);

      timerInterval = setInterval(function() {
        var now = new Date().getTime();
        var distance = targetTime - now;

        var currDate = new Date();
        document.getElementById('currentTimeText').innerText = currDate.toLocaleString('ko-KR');

        if (distance <= 0) {
          clearInterval(timerInterval);
          document.getElementById('countdownTimer').innerText = "00:00:00";
          document.getElementById('loading').style.display = 'block';
          document.getElementById('loading').innerText = "시간이 되었습니다! 시간표를 여는 중... 🚀";
          fetchData(); 
          return;
        }

        var days = Math.floor(distance / (1000 * 60 * 60 * 24));
        var hours = Math.floor((distance % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
        var minutes = Math.floor((distance % (1000 * 60 * 60)) / (1000 * 60));
        var seconds = Math.floor((distance % (1000 * 60)) / 1000);

        hours = hours < 10 ? "0" + hours : hours;
        minutes = minutes < 10 ? "0" + minutes : hours; // 기존 버그 수정 (minutes 매핑)
        seconds = seconds < 10 ? "0" + seconds : seconds;

        var timerText = "";
        if (days > 0) {
          timerText = days + "일 " + hours + ":" + minutes + ":" + seconds;
        } else {
          timerText = hours + ":" + minutes + ":" + seconds;
        }

        document.getElementById('countdownTimer').innerText = timerText;
      }, 1000);
    }

    function searchName() {
      var input = document.getElementById("searchInput").value.trim();
      var cells = document.getElementsByClassName("searchable");
      for (var i = 0; i < cells.length; i++) {
        var cell = cells[i];
        cell.classList.remove("highlight");
        if (input !== "" && cell.innerHTML.indexOf(input) !== -1) {
          cell.classList.add("highlight");
        }
      }
    }
