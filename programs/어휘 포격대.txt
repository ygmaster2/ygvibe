/**
 * 웹앱이 실행 중인 스프레드시트에서 데이터를 자동으로 추출하고 무작위 10문항을 고르는 함수
 * 160개 이상의 문장을 안전하게 가져와 완벽한 무작위성을 보장합니다.
 */
function getQuizData() {
  try {
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    if (!ss) {
      throw new Error("컨테이너 스프레드시트를 찾을 수 없습니다. 시트 내부에서 실행해주세요.");
    }
    
    const sheet = ss.getSheets()[0]; 
    const lastRow = sheet.getLastRow();
    
    // 시트에 데이터가 전혀 없을 경우 안전하게 모의 데이터 반환
    if (lastRow < 2) {
      return getSampleData();
    } 
    
    // 9열까지의 모든 데이터를 한 번에 가져옵니다.
    const range = sheet.getRange(2, 1, lastRow - 1, 9);
    const values = range.getValues();
    
    // 1. 본문 데이터가 실제로 존재하는 유효한 문항만 엄격하게 필터링
    let allQuizList = values
      .filter(function(row) {
        return row[2] && row[2].toString().trim() !== "";
      })
      .map(function(row, index) {
        let ansNum = parseInt(row[7]);
        if (isNaN(ansNum) || ansNum < 1) {
          ansNum = 1;
        }

        return {
          id: row[0] ? row[0].toString() : String(index + 1), 
          source: row[1] ? row[1].toString() : "EBS 수특",  
          sentence: row[2].toString().trim(),                  
          options: [
            row[3] ? row[3].toString().trim() : "보기1",
            row[4] ? row[4].toString().trim() : "보기2",
            row[5] ? row[5].toString().trim() : "보기3",
            row[6] ? row[6].toString().trim() : "보기4"
          ], 
          originalAnswerIdx: ansNum - 1,                        
          feedback: row[8] ? row[8].toString().trim() : "정답입니다! 문맥상 알맞은 어휘입니다." 
        };
      });
    
    if (allQuizList.length === 0) {
      return getSampleData();
    }
    
    // 2. 전체 리스트를 피셔-예이츠 알고리즘으로 섞음
    for (let i = allQuizList.length - 1; i > 0; i--) {
      const j = Math.floor(Math.random() * (i + 1));
      const temp = allQuizList[i];
      allQuizList[i] = allQuizList[j];
      allQuizList[j] = temp;
    }
    
    // 3. 무작위로 완벽하게 섞인 전체 범위에서 최대 10문항만 추출
    const selectedQuizCount = Math.min(10, allQuizList.length);
    const finalQuizList = allQuizList.slice(0, selectedQuizCount);
    
    return finalQuizList;
  } catch (e) {
    Logger.log("Error in getQuizData: " + e.toString());
    throw new Error(e.message || e.toString());
  }
}

/**
 * 랭킹 등록 시스템 시트 확인 및 자동 레이아웃 설정
 * 지정된 구글 시트에 'Ranking' 탭을 개설하고 구조를 보정합니다.
 */
function getRankingSheet() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let rankSheet = ss.getSheetByName("Ranking");
  if (!rankSheet) {
    // 랭킹 시트가 없을 경우 성벽 파손율 열을 포함하여 자동 개설
    rankSheet = ss.insertSheet("Ranking");
    rankSheet.appendRow(["등록 일시", "플레이어 이름", "누적 스코어", "성벽 파손율"]);
    // 열 디자인 서식 설정
    rankSheet.getRange("A1:D1").setFontWeight("bold").setBackground("#1e293b").setFontColor("#ffffff");
    rankSheet.setColumnWidth(1, 180);
    rankSheet.setColumnWidth(2, 150);
    rankSheet.setColumnWidth(3, 120);
    rankSheet.setColumnWidth(4, 120);
  }
  return rankSheet;
}

/**
 * 신규 랭킹 스코어 및 성벽 파손율 저장 함수
 * @param {string} name - 플레이어 이니셜
 * @param {number} score - 획득한 누적 점수
 * @param {number} damage - 최종 성벽 파손율 (정수값)
 */
function submitScore(name, score, damage) {
  try {
    const sheet = getRankingSheet();
    const cleanName = name ? name.toString().trim().substring(0, 10) : "AAA";
    const cleanScore = parseInt(score);
    const cleanDamage = parseInt(damage) || 0; // 성벽 파손율 수치 파싱 보완
    const now = new Date();
    const formattedDate = Utilities.formatDate(now, Session.getScriptTimeZone(), "yyyy-MM-dd HH:mm:ss");
    
    // 데이터 줄 마지막 행에 추가 기재 (성벽 파손율을 4번째 열에 배치)
    sheet.appendRow([formattedDate, cleanName, cleanScore, cleanDamage]);
    return { success: true };
  } catch(e) {
    Logger.log("Error in submitScore: " + e.toString());
    return { success: false, error: e.toString() };
  }
}

/**
 * 상위 100명의 랭킹 데이터를 내림차순 정렬하여 읽어오는 함수
 * 파손율을 포함하여 클라이언트 측으로 전송합니다.
 */
function getTopRankings() {
  try {
    const sheet = getRankingSheet();
    const lastRow = sheet.getLastRow();
    if (lastRow < 2) {
      return [];
    }
    
    // 4열(성벽 파손율)까지로 로드 영역 확장
    const range = sheet.getRange(2, 1, lastRow - 1, 4);
    const values = range.getRangeValues ? range.getRangeValues() : range.getValues();
    
    // 스코어 기준으로 완벽한 내림차순 정렬 수행
    const sortedData = values.map(function(row) {
      return {
        date: row[0] ? Utilities.formatDate(new Date(row[0]), Session.getScriptTimeZone(), "MM/dd HH:mm") : "",
        name: row[1] ? row[1].toString() : "AAA",
        score: parseInt(row[2]) || 0,
        damage: row[3] !== undefined ? parseInt(row[3]) : 0 // 기존 파손율 데이터가 존재하지 않던 플레이어는 0% 기본 보정
      };
    }).sort(function(a, b) {
      return b.score - a.score;
    });
    
    // 최대 상위 100명의 랭킹 배열 분할 반환
    return sortedData.slice(0, 100);
  } catch(e) {
    Logger.log("Error in getTopRankings: " + e.toString());
    return [];
  }
}

  function startBgmOnFirstInteract() {
    if(!isMuted) {
      changeVolume(currentVolume);

    }
  }

  function toggleMute() {
    isMuted = !isMuted;
    const muteBtn = document.getElementById('mute-btn');
    
    if (isMuted) {
      muteBtn.innerHTML = '<i class="fa-solid fa-volume-xmark" style="color:var(--neon-red);"></i>';
      Object.keys(audioAssets).forEach(key => {
        audioAssets[key].pause();
      });
    } else {
      muteBtn.innerHTML = '<i class="fa-solid fa-volume-high"></i>';
      changeVolume(currentVolume);
      audioAssets.bgm.play().catch(() => {});
    }
  }

  function changeVolume(val) {
    currentVolume = parseFloat(val);
    
    if(isMuted) {
      isMuted = false;
      document.getElementById('mute-btn').innerHTML = '<i class="fa-solid fa-volume-high"></i>';
    }
    
    audioAssets.bgm.volume = currentVolume * 0.15;
    audioAssets.correct.volume = currentVolume * 0.8;
    audioAssets.incorrect.volume = currentVolume * 0.5;
    audioAssets.victory.volume = currentVolume * 0.7;
  }

    if(isMuted) return;
    
    const sfx = audioAssets[type];
    if(sfx) {
      changeVolume(currentVolume);
      sfx.currentTime = 0;
      sfx.play().catch(e => console.log("SFX 재생 차단 회피: " + e));
    }
  }

  function initStage() {
    if (!quizData || quizData.length === 0 || currentIdx >= quizData.length) {
      gameOver();

    let formattedSentence = rawSentence.replace(/\[\s?\?\s?\]/g, 
      `<span style="color:var(--neon-gold); text-shadow:0 0 10px rgba(251,191,36,0.6); font-weight:900; display:inline-block; white-space:nowrap; padding: 0 4px;">[ ? ]</span>`
    );
    
    document.getElementById('quiz-box').innerHTML = formattedSentence;
    
    const rawOptions = [...currentQuiz.options];
    const correctText = rawOptions[currentQuiz.originalAnswerIdx];
    
    currentShuffledOptions = rawOptions.map((text, idx) => ({ text, originalIdx: idx }));
    for (let i = currentShuffledOptions.length - 1; i > 0; i--) {
      const j = Math.floor(Math.random() * (i + 1));
      [currentShuffledOptions[i], currentShuffledOptions[j]] = [currentShuffledOptions[j], currentShuffledOptions[i]];
    }
    
    currentCorrectIndexInShuffle = currentShuffledOptions.findIndex(item => item.text === correctText);

  /* 🎆 축하 및 완전 격파 시 승리의 폭죽 이펙트 트리거 함수 */
  function triggerVictoryFireworks(intensity = 15) {
    const container = document.getElementById('fortress-wall');
    const colors = ['#38bdf8', '#fbbf24', '#f43f5e', '#10b981', '#a855f7', '#ff7849'];
    
    // 강도(intensity)에 따른 폭죽 다발 생성
    for (let i = 0; i < intensity; i++) {
      setTimeout(() => {
        const fireworkCount = 10;
        const centerX = 20 + Math.random() * 60; // 20% ~ 80% 가로축
        const centerY = 30 + Math.random() * 40; // 30% ~ 70% 세로축
        const randomColor = colors[Math.floor(Math.random() * colors.length)];

        for (let j = 0; j < fireworkCount; j++) {
          const particle = document.createElement('div');
          particle.className = 'victory-firework';
          particle.style.left = `${centerX}%`;
          particle.style.top = `${centerY}%`;
          particle.style.background = randomColor;
          particle.style.boxShadow = `0 0 12px ${randomColor}, 0 0 24px ${randomColor}`;

          const angle = (j / fireworkCount) * Math.PI * 2 + (Math.random() * 0.4);
          const distance = 40 + Math.random() * 70;
          const tx = Math.cos(angle) * distance;
          const ty = Math.sin(angle) * distance;

          particle.style.setProperty('--tx', `${tx}px`);
          particle.style.setProperty('--ty', `${ty}px`);

          container.appendChild(particle);
          setTimeout(() => particle.remove(), 800);
        }
      }, i * 180);
    }
  }

  function fireShell(chosenIdx) {
    if(isAnswered || isPenaltyState) return;

    const buttons = document.querySelectorAll('.ammo-btn');

    if (chosenIdx === currentCorrectIndexInShuffle) {
      isAnswered = true;
      clearInterval(timerInterval);
      buttons.forEach(btn => btn.disabled = true); // 포격 연출 중 추가 클릭 봉쇄

      animateCannonballFire(() => {
        const prevDamage = totalWallDamage;
        totalWallDamage += damagePerStage;
        if (totalWallDamage > 100) {
          totalWallDamage = 100;
        }

        let earnedBonus = 0;
        let bonusTierReached = 0;

        // 💥 [개정] 성벽 파손율 구간 점수 계산 알고리즘 (60%~100% 구간 보너스 획득 감지)
        const currentDamageCeil = Math.ceil(totalWallDamage);
        
        if (currentDamageCeil >= 100 && lastBonusTier < 100) {
          bonusTierReached = 100;
          earnedBonus = 5000;
          lastBonusTier = 100;
        } else if (currentDamageCeil >= 90 && lastBonusTier < 90) {
          bonusTierReached = 90;
          earnedBonus = 4000;
          lastBonusTier = 90;
        } else if (currentDamageCeil >= 80 && lastBonusTier < 80) {
          bonusTierReached = 80;
          earnedBonus = 3000;
          lastBonusTier = 80;
        } else if (currentDamageCeil >= 70 && lastBonusTier < 70) {
          bonusTierReached = 70;
          earnedBonus = 2000;
          lastBonusTier = 70;
        } else if (currentDamageCeil >= 60 && lastBonusTier < 60) {
          bonusTierReached = 60;
          earnedBonus = 1000;
          lastBonusTier = 60;
        }

        // 보너스 점수 적립 및 화려한 승리 효과 트리거
        if (earnedBonus > 0) {
          score += earnedBonus;
          
          // 파괴 구간별 축포 강도 설정 (보너스가 클수록 폭죽 다발 스폰 확대)
          const fireworkIntensity = bonusTierReached === 100 ? 18 : Math.floor(bonusTierReached / 10);
          triggerVictoryFireworks(fireworkIntensity);
        }

        renderWallVisual();
        playSound('correct');

        const viewport = document.getElementById('game-viewport');
        viewport.classList.add('shake');
        
        setTimeout(() => {
          viewport.classList.remove('shake');
          
          // 구간 보너스 획득에 따른 알림 전용 텍스트 조율
          if (earnedBonus > 0) {
            if (bonusTierReached === 100) {
              showFeedbackModal(true, `🏰 성채 완전 함락!! 최종 완파 보너스 [ +5,000 PTS ] 적립 완료! 🎉`);
            } else {
              showFeedbackModal(true, `⚔️ 전술적 돌파 성공!! 성벽 파손 ${bonusTierReached}% 보너스 [ +${earnedBonus.toLocaleString()} PTS ] 적립 완료! 🎉`);
            }
          } else {
            showFeedbackModal(true);
          }
        }, 400);

        let damageScore = Math.max(25, Math.floor(currentTime));
        score += damageScore * 10;
        document.getElementById('score-display').innerText = String(score).padStart(5, '0');
      });
    } else {
      triggerPenaltyLock(chosenIdx);
    }
  }

  function triggerPenaltyLock(failedIdx) {
    isPenaltyState = true;
    
    const buttons = document.querySelectorAll('.ammo-btn');
    buttons[failedIdx].classList.add('ammo-destroyed');
    buttons.forEach(btn => btn.disabled = true);

    triggerMisfireEffect();
    playSound('incorrect');

    currentTime = Math.max(0, currentTime / 2);
    document.getElementById('time-bar').style.height = `${currentTime}%`;

    const timeBar = document.getElementById('time-bar');
    timeBar.classList.add('slashed');
    setTimeout(() => {
      timeBar.classList.remove('slashed');
    }, 200);
    
    document.getElementById('penalty-alert-text').style.display = "block";
    document.getElementById('game-viewport').classList.add('penalty-shake');
    
    setTimeout(function() {
      isPenaltyState = false;
      document.getElementById('penalty-alert-text').style.display = "none";
      document.getElementById('game-viewport').classList.remove('penalty-shake');
      
      if (currentTime > 0) {
        buttons.forEach(btn => {
          if(!btn.classList.contains('ammo-destroyed')) {
            btn.disabled = false;
          }
        });
      }
    }, 1000);
  }

  function handleTimeOutBreak() {
    isAnswered = true;
    playSound('incorrect');

    const buttons = document.querySelectorAll('.ammo-btn');
    buttons.forEach(btn => btn.disabled = true);
    
    const viewport = document.getElementById('game-viewport');
    viewport.classList.add('shake');
    setTimeout(() => {
      viewport.classList.remove('shake');
      showFeedbackModal(false); 
    }, 400);
  }

  function showFeedbackModal(isSuccess, extraMessage = '') {
    const currentQuiz = quizData[currentIdx];
    const modal = document.getElementById('modal');
    
    document.getElementById('ranking-section').style.display = 'none';
    document.getElementById('btn-retry').style.display = 'none';
    document.getElementById('modal-btn').style.display = 'inline-block';

    if(isSuccess) {
      document.getElementById('modal-title').innerText = "WALL CRUSHED! ⚔️";
      document.getElementById('modal-title').style.color = "var(--neon-green)";
    } else {
      document.getElementById('modal-title').innerText = "TIME OVER! DEFENSE HELD 🛡️";
      document.getElementById('modal-title').style.color = "var(--neon-red)";
    }
    
    document.getElementById('modal-source').innerText = `SOURCE: ${currentQuiz.source}`;
    
    // 추가 메시지(예: 구간 보너스 획득 소식) 결합
    let feedbackContent = currentQuiz.feedback;
    if (extraMessage) {
      feedbackContent = `${extraMessage}\n\n${feedbackContent}`;
    }
    document.getElementById('modal-feedback').innerText = feedbackContent;
    
    if(currentIdx === quizData.length - 1) {
      document.getElementById('modal-btn').innerText = "FINAL OPERATION REPORT 🏆";
    } else {
      document.getElementById('modal-btn').innerText = "NEXT ATTACK MOVE ⚔️";
    }
    
    modal.style.display = 'flex';
  }

  function nextStage() {

    initStage();
  }

  function gameOver() {
    clearInterval(timerInterval);
    const modal = document.getElementById('modal');
    
    audioAssets.bgm.pause();
    playSound('victory');

    let finishTitle = totalWallDamage >= 95 ? "COMPLETE TACTICAL VICTORY! 🏆" : "OPERATION CONCLUDED ⚔️";
    
    document.getElementById('modal-title').innerText = finishTitle;
    document.getElementById('modal-title').style.color = "var(--neon-gold)";
    document.getElementById('modal-source').innerText = "MISSION REPORT";
    document.getElementById('modal-feedback').innerText = `모든 공성 포격 작전이 마감되었습니다.\n\n최종 성벽 파손율: ${Math.ceil(totalWallDamage)}%\n누적 최종 스코어: ${score} PTS`;
    
    document.getElementById('modal-btn').style.display = 'none';
    document.getElementById('btn-retry').style.display = 'inline-block';
    
    document.getElementById('ranking-section').style.display = 'block';
    hasSubmitted = false;
    document.getElementById('player-name-input').disabled = false;
    document.getElementById('btn-submit-score').disabled = false;
    document.getElementById('ranking-status').style.display = 'none';
    
    modal.style.display = 'flex';

    fetchRankings();
  }

  function fetchRankings() {
    document.getElementById('ranking-loader').style.display = 'block';
    document.getElementById('ranking-table').style.display = 'none';

    if (typeof google !== 'undefined' && google.script && google.script.run) {
      google.script.run
        .withSuccessHandler(function(rankings) {
          const tbody = document.getElementById('ranking-list');
          tbody.innerHTML = '';
          
          if(rankings.length === 0) {
            tbody.innerHTML = '<tr><td colspan="5" style="text-align:center; color:#94a3b8;">아직 등록된 명예로운 스코어가 없습니다. 첫 도전을 새겨보세요!</td></tr>';
          } else {
            rankings.forEach((player, index) => {
              const tr = document.createElement('tr');
              
              let rankStyle = '';
              if (index === 0) rankStyle = 'style="color:#ffd700; font-weight:bold;"';
              else if (index === 1) rankStyle = 'style="color:#c0c0c0; font-weight:bold;"';
              else if (index === 2) rankStyle = 'style="color:#cd7f32; font-weight:bold;"';

              const isRecentUser = !hasSubmitted && player.name === document.getElementById('player-name-input').value.toUpperCase() && player.score === score;
              if (isRecentUser) {
                tr.className = 'rank-highlight';
              }

              const displayDmg = player.damage !== undefined ? player.damage : 0;

              tr.innerHTML = `
                <td class="rank-num" ${rankStyle}>${index + 1}</td>
                <td>${escapeHtml(player.name)}</td>
                <td style="color:var(--neon-blue); font-weight:bold;">${player.score.toLocaleString()}</td>
                <td style="color:var(--neon-red); font-weight:bold;">${displayDmg}%</td>
                <td style="font-size:11px; color:#64748b;">${player.date}</td>
              `;
              tbody.appendChild(tr);
            });
          }
          
          document.getElementById('ranking-loader').style.display = 'none';
          document.getElementById('ranking-table').style.display = 'table';
        })
        .withFailureHandler(function() {
          const tbody = document.getElementById('ranking-list');
          tbody.innerHTML = '<tr><td colspan="5" style="text-align:center; color:var(--neon-red);">랭킹 데이터 로딩 실패</td></tr>';
          document.getElementById('ranking-loader').style.display = 'none';
          document.getElementById('ranking-table').style.display = 'table';
        })
        .getTopRankings();
    } else {
      document.getElementById('ranking-loader').style.display = 'none';
      document.getElementById('ranking-table').style.display = 'table';
      const tbody = document.getElementById('ranking-list');
      tbody.innerHTML = '<tr><td colspan="5" style="text-align:center; color:#94a3b8;">[로컬 환경] 명예의 전당은 구글 웹앱 실행 상태에서 작동합니다.</td></tr>';
    }
  }

  function submitPlayerScore() {
    const nameInput = document.getElementById('player-name-input');
    const pName = nameInput.value.trim().toUpperCase();
    
    if(!pName) {
      alertCustom("플레이어 이름을 입력해 주세요!");
      return;
    }
    
    if(hasSubmitted) return;

    const statusText = document.getElementById('ranking-status');
    statusText.innerText = "⚡ 명예의 전당 장막에 기록을 새기는 중...";
    statusText.style.color = "var(--neon-gold)";
    statusText.style.display = "block";
    
    nameInput.disabled = true;
    document.getElementById('btn-submit-score').disabled = true;

    const currentDmgRate = Math.ceil(totalWallDamage);

    if (typeof google !== 'undefined' && google.script && google.script.run) {
      google.script.run
        .withSuccessHandler(function(res) {
          if(res.success) {
            statusText.innerText = "🏆 기록 등재 완료! 최상단 랭킹보드가 즉시 반영됩니다.";
            statusText.style.color = "var(--neon-green)";
            hasSubmitted = true;
            fetchRankings();
          } else {
            statusText.innerText = "❌ 저장 오류: " + res.error;
            statusText.style.color = "var(--neon-red)";
            nameInput.disabled = false;
            document.getElementById('btn-submit-score').disabled = false;
          }
        })
        .withFailureHandler(function(err) {
          statusText.innerText = "❌ 구글 통신 장애 발생: " + err.message;
          statusText.style.color = "var(--neon-red)";
          nameInput.disabled = false;
          document.getElementById('btn-submit-score').disabled = false;
        })
        .submitScore(pName, score, currentDmgRate);
    } else {
      statusText.innerText = "⚠️ 연동되지 않은 가상 환경에서는 점수를 등록할 수 없습니다.";
      statusText.style.color = "var(--neon-red)";
    }
  }

  function alertCustom(text) {
    const statusText = document.getElementById('ranking-status');
    statusText.innerText = "⚠️ " + text;
    statusText.style.color = "var(--neon-red)";
    statusText.style.display = "block";
  }

  function escapeHtml(str) {
    return str
      .replace(/&/g, "&amp;")
      .replace(/</g, "&lt;")
      .replace(/>/g, "&gt;")
      .replace(/"/g, "&quot;")
      .replace(/'/g, "&#039;");
  }
